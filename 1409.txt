using System;

namespace ConsoleApplication1
{
    class Program
    {
        static void Main(string[] args)
        {
            var s = Console.ReadLine().Split();
            int k = int.Parse(s[0]) + int.Parse(s[1]) - 1;
            Console.WriteLine((k - int.Parse(s[0])) + " " + (k - int.Parse(s[1])));
        }
    }
}
